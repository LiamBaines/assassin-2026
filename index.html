    const assassin = params.get("assassin");

    const target = params.get("target");


    const greetingEl = document.getElementById("greeting");

    const targetEl = document.getElementById("target");


    if (!assassin || !target) {

      greetingEl.innerText = "INVALID BRIEFING";

      targetEl.innerText = "Missing parameters.";

    } else {

      greetingEl.innerText = `Hello ${assassin}.`;

      targetEl.innerText = `Your target is ${target}.`;

    }
